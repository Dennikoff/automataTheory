function QuickSort(vector of int arr, int n)
	begin
	int i;
	int j;
	int sw set i;
	int min set i;
	do
		begin
		min set i;
		j set (i add 1);
		do
			begin
			if arr[j] first smaller arr[min] then
			    min set j;
			else;
			j set (j add 1);
			end
		while j first smaller n;
		sw set arr[i];
		arr[i] set arr[min];
		arr[min] set sw;
		i set (i add 1);
		end
	while i first smaller (n sub 1);
	end
return arr;
